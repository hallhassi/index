const fs = require('fs'); const { createCanvas, loadImage } = require('canvas'); const path = require('path'); // Canvas dimensions (The actual stitched sheets) const MAX_W = 300; const MAX_H = 400; const COLS = 4; const BATCH_SIZES = [12, 48, 48, 42]; const SOURCE_DIR = './thumbnails'; async function run() { // Filter out hidden files (.DS_Store) and non-images const files = fs.readdirSync(SOURCE_DIR) .filter(f => !f.startsWith('.') && f.match(/\.(jpg|jpeg|png|webp)$/i)) .sort(); if (files.length > 0) { console.log(`First file detected: "${files[0]}"`); } else { console.error("No images found in the directory!"); return; } // Table width: 4 columns * (400px img + 400px text) = 3200px let htmlContent = ''; let filePointer = 0; let globalImageCount = 0; for (let s = 0; s < BATCH_SIZES.length; s++) { const count = BATCH_SIZES[s]; const rows = Math.ceil(count / COLS); const actualCount = Math.min(count, files.length - filePointer); if (actualCount <= 0) break; const canvas = createCanvas(COLS * MAX_W, rows * MAX_H); const ctx = canvas.getContext('2d'); ctx.fillStyle = 'white'; ctx.fillRect(0, 0, canvas.width, canvas.height); for (let i = 0; i < actualCount; i++) { const fileName = files[filePointer]; const imgPath = path.join(SOURCE_DIR, fileName); try { const img = await loadImage(imgPath); // SHEET DRAWING LOGIC const ratio = Math.min(MAX_W / img.width, MAX_H / img.height); const drawW = img.width * ratio; const drawH = img.height * ratio; const cellX = (i % COLS) * MAX_W; const cellY = Math.floor(i / COLS) * MAX_H; const offsetX = cellX + (MAX_W - drawW) / 2; const offsetY = cellY + (MAX_H - drawH) / 2; ctx.drawImage(img, offsetX, offsetY, drawW, drawH); // HTML TABLE LOGIC if (globalImageCount % COLS === 0) { htmlContent += ''; } // Image Cell (400px) htmlContent += ``; // Filename Cell (400px) htmlContent += ``; if (globalImageCount % COLS === 3) { htmlContent += ''; } globalImageCount++; } catch (err) { console.error(`Skipping file ${fileName} due to error: ${err.message}`); } filePointer++; } // Save sheet as JPG 70% const out = fs.createWriteStream(`sheet_${s + 1}.jpg`); const stream = canvas.createJPEGStream({ quality: 0.7 }); stream.pipe(out); await new Promise((res) => out.on('finish', res)); console.log(`Successfully saved sheet_${s + 1}.jpg`); } // Close row if it ends mid-column if (globalImageCount % COLS !== 0) { htmlContent += ''; } htmlContent += '
`; htmlContent += `${fileName}
'; const finalHtml = ` scanner ${htmlContent} `; fs.writeFileSync('index.html', finalHtml); console.log('Successfully generated index.html'); } run().catch(console.error);
